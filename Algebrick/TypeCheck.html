def a_type_check(which, bang, value, *types)
  ok = types.any? do |t|
    case which
    when :===
      t === value
    when :kind_of?
      value.kind_of? t
    else
      raise ArgumentError
    end
  end
  bang && !ok and
      raise TypeError,
            "value (#{value.class}) '#{value}' is not ##{which} any of #{types.join(', ')}"
  bang ? value : ok
end
