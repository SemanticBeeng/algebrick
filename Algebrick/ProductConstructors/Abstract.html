def initialize(*fields)
  if fields.size == 1 && fields.first.is_a?(Hash)
    fields = type.field_names.map { |k| fields.first[k] }
  end
  @fields = fields.zip(self.class.type.fields).map { |field, type| Type! field, type }.freeze
end
